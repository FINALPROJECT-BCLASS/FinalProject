DROP TABLE GROUP_PLAN;

CREATE TABLE GROUP_PLAN (
	GP_NO	NUMBER		NOT NULL,
	G_NO	NUMBER		NOT NULL,
	GM_NO	NUMBER		NOT NULL,
	GP_TITLE 	VARCHAR2(100)		NULL,
	GP_CON	VARCHAR2(100)		NULL,
	GP_START	VARCHAR2(100)		NULL,
	GP_END	VARCHAR2(100)		NULL,
	GP_COORDY VARCHAR2(500) NULL,
    GP_COORDX VARCHAR2(500) NULL,
    GP_ADDRESS1 VARCHAR2(500) NULL,
    GP_ADDRESS2 VARCHAR2(500) NULL,
	GP_COLOR	VARCHAR2(100)		NULL,
	GP_DELETE	VARCHAR2(2)	DEFAULT 'N'	NULL
);

COMMENT ON COLUMN GROUP_PLAN.GP_NO IS '번호';

COMMENT ON COLUMN GROUP_PLAN.G_NO IS '그룹 번호';

COMMENT ON COLUMN GROUP_PLAN.GM_NO	 IS '그룹 멤버 번호';

COMMENT ON COLUMN GROUP_PLAN.GP_TITLE  IS '제목';

COMMENT ON COLUMN GROUP_PLAN.GP_CON IS '내용';

COMMENT ON COLUMN GROUP_PLAN.GP_START IS '시작날짜';

COMMENT ON COLUMN GROUP_PLAN.GP_END IS '종료날짜';

COMMENT ON COLUMN GROUP_PLAN.GP_COORDY IS '위도';

COMMENT ON COLUMN GROUP_PLAN.GP_COORDX IS '경도';

COMMENT ON COLUMN GROUP_PLAN.GP_ADDRESS1 IS '기본주소';

COMMENT ON COLUMN GROUP_PLAN.GP_ADDRESS2 IS '상세주소';

COMMENT ON COLUMN GROUP_PLAN.GP_COLOR IS '색상';

COMMENT ON COLUMN GROUP_PLAN.GP_DELETE IS '삭제 여부';
